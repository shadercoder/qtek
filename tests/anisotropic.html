require(['qtek'], function(qtek){ var qtek3d = qtek['3d'] var Shader = qtek3d.Shader var _ = require("_"); var renderer = new qtek3d.Renderer({ canvas : document.getElementById( "Main"), devicePixelRatio : 1.0 }); var scene = new qtek3d.Scene, camera = new qtek3d.camera.Perspective({ aspect : renderer.canvas.width/renderer.canvas.height, far : 500 }); var shader = qtek3d.shader.library.get("buildin.basic", "diffuseMap"); var material = new qtek3d.Material({ shader : shader }); var image = new Image; image.src = "assets/textures/ground_tile.jpg"; var texture = new qtek3d.texture.Texture2D({ image : image, wrapS : qtek3d.Texture.REPEAT, wrapT : qtek3d.Texture.REPEAT, anisotropic : 16 }); image.onload = function(){ texture.dirty(); }; material.set("diffuseMap", texture); material.set("uvRepeat", [10, 10]) var root = new qtek3d.Node(); camera.position.set(0, 4, 14); camera.lookAt( new qtek.core.Vector3(0, 1, 0) ); scene.add( root ); // Add Plane var plane = new qtek3d.geometry.Plane({ widthSegments : 1, heightSegments : 1 }); var planeMesh = new qtek3d.Mesh({ geometry : plane, material : material, scale : new qtek.core.Vector3(60, 60, 60) }); planeMesh.position.y = -0.8; planeMesh.rotation.rotateX(-Math.PI/2); root.add(planeMesh); setInterval(function(){ renderer.render( scene, camera ); }, 20); })
